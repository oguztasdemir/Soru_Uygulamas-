\end{learningcard}

\begin{learningcard}{SOM (Kendi Kendini Organize Eden Haritalar) sinir ağı mimarisinin temel işlevi nedir?}
Yüksek boyutlu veriyi topolojiyi koruyarak düşük boyutlu (genellikle 2D) bir haritaya indirgemek.
\end{learningcard}

\begin{learningcard}{SVD (Tekil Değer Ayrışımı) ile PCA arasındaki ilişki nedir?}
SVD, PCA hesaplamasını sayısal olarak daha kararlı bir şekilde gerçekleştirmek için kullanılır.
\end{learningcard}

\begin{learningcard}{SVM algoritmasında 'Soft Margin' kullanımı neyi amaçlar?}
Verideki bazı gürültülü noktaların yanlış sınıfta kalmasına izin vererek daha genel bir karar sınırı oluşturmayı.
\end{learningcard}

\begin{learningcard}{SVM'de 'Kernel Trick' ne işe yarar?}
Veriyi daha yüksek boyutlu bir özellik uzayına taşıyarak doğrusal olmayan sınırların çizilmesini sağlar.
\end{learningcard}

\begin{learningcard}{SVM'de veriyi doğrusal olmayan bir üst uzaya taşıyarak ayrıştırmayı sağlayan yönteme ne denir?}
Kernel Trick (Çekirdek Yöntemi).
\end{learningcard}

\begin{learningcard}{Sınıflandırma (Classification) problemi makine öğrenmesinin hangi kategorisine girer?}
Denetimli Öğrenme (Supervised Learning).
\end{learningcard}

\begin{learningcard}{Sınıflandırma performansını ölçen Karmaşıklık Matrisi'nde (Confusion Matrix) FN neyi temsil eder?}
Yanlış Negatif (Gerçekte pozitif olanın negatif tahmin edilmesi).
\end{learningcard}

\begin{learningcard}{Sınıflandırma probleminde modelin çıktı olarak ürettiği olasılığı 0 veya 1'e dönüştüren fonksiyona ne denir?}
Sigmoid Fonksiyonu: $\sigma(z) = \frac{1}{1 + e^{-z}}$.
\end{learningcard}

\begin{learningcard}{t-SNE algoritması yüksek boyuttaki benzerlikleri düşük boyuta taşırken hangi olasılık dağılımını kullanır?}
t-dağılımı (Student's t-distribution).
\end{learningcard}

\begin{learningcard}{UMAP algoritmasının t-SNE'ye göre en büyük pratik avantajı nedir?}
Daha hızlı çalışması ve yeni veri noktalarının projeksiyonuna izin vermesi.
\end{learningcard}

\begin{learningcard}{UMAP yönteminin t-SNE'ye göre avantajı nedir?}
Daha hızlı çalışması ve yeni veri noktalarını mevcut projeksiyona dahil edebilmesidir.
\end{learningcard}

\begin{learningcard}{Vektörleştirme (Vectorization) sinir ağı eğitiminde neden önemlidir?}
Çok sayıda skaler işlemi tek bir matris çarpımına dönüştürerek donanım (GPU) hızlandırması sağlar.
\end{learningcard}

\begin{learningcard}{Veriyi 2D veya 3D'de görselleştirmek için t-SNE'nin kullandığı temel benzerlik ölçütü nedir?}
Koşullu olasılıklar tabanlı benzerlik (t-dağılımı).
\end{learningcard}

\begin{learningcard}{XOR problemi neden tek katmanlı (doğrusal) modellerle çözülemez?}
XOR verisi doğrusal olarak ayrıştırılamaz, modellemek için doğrusal olmayan bir gizli katman gerekir.
\end{learningcard}

\begin{learningcard}{Yapay sinir ağlarında 'Dropout' yönteminin işlevi nedir?}
Eğitim sırasında nöronları rastgele kapatarak modelin belirli nöronlara aşırı bağımlı olmasını (overfitting) engellemek.
\end{learningcard}

\begin{learningcard}{Yapay sinir ağlarında bir nöronun temel matematiksel denklemi nedir?}
$a = \phi(\sum w_j x_j + b)$ (Burada $\phi$ aktivasyon fonksiyonudur).
\end{learningcard}

\begin{learningcard}{Yapay sinir ağlarında tek bir nöronun matematiksel denklemi nedir?}
$a = \phi(\sum w_j x_j + b)$
\end{learningcard}

\begin{learningcard}{Yüksek boyutlu veriyi 2D veya 3D uzayda görselleştirmek için en sık tercih edilen doğrusal olmayan yöntemler hangileridir?}
t-SNE ve UMAP yöntemleri.
\end{learningcard}

\begin{learningcard}{Çok sınıflı sınıflandırma problemlerinde kullanılan hata fonksiyonu hangisidir?}
Çapraz Entropi (Cross-Entropy Loss).
\end{learningcard}

\begin{learningcard}{Özelliklerin dönüştürülerek yeni yapay bileşenlerin üretildiği işleme ne ad verilir?}
Boyut Düşürme (Dimensionality Reduction).
\end{learningcard}
